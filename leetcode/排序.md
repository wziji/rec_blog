# 排序算法

### 1. 快排

```python
def quick_sort(input_list):
	if len(input_list) < 2:
		return input_list

	base_value = input_list[0]
	less_list = []
	more_list = []

	for v in input_list[1:]:
		if v > base_value:
			more_list.append(v)
		else:
			less_list.append(v)

	return quick_sort(less_list) + [base_value] + quick_sort(more_list)
```

### 2. 归并排序

```python
def sort(input_list):
	if len(input_list) < 2:
		return input_list

	mid = len(input_list) // 2

	left = sort(input_list[:mid])
	right = sort(input_list[mid:])

	return merge(left, right)


def merge(left, right):
	result_list = []

	while left and right:
		if left[0] > right[0]:
			result_list.append(right[0])
			right = right[1:]

		else:
			result_list.append(left[0])
			left = left[1:]

	result_list += left
	result_list += right

	return result_list
```
